Configure Keyboard

4 5

cp ~/.termux/termux.properties ~/.termux/termux.properties.bak
vim ~/.termux/termux.properties

extra-keys = [[ \
  {key: TAB, popup: KEYBOARD}, \
  {key: ESC, popup: '<'}, \
  {key: CTRL, popup: '['}, \
  {key: ALT, popup: '\{'}, \
  {key: 'BACKSLASH', popup: '|'}, \
  {key: '_', popup: '='}, \
  {key: UP, popup: PGUP}, \
  {key: DOWN, popup: PGDN}, \
  {key: LEFT, popup: HOME}, \
  {key: RIGHT, popup: END} \
]]
